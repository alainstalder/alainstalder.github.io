package ch.grengine.jexler.service

import groovy.transform.CompileStatic

/**
 * Cron service event.
 *
 * @author Jex Jexler (Alain Stalder)
 */
@CompileStatic
class CronEvent extends EventBase {

    private final String cron

    /**
     * Constructor.
     * @param service the service that created the event
     * @param cron the cron pattern that caused the event
     */
    CronEvent(final Service service, final String cron) {
        super(service)
        this.cron = cron
    }

    /**
     * Get cron pattern.
     */
    String getCron() {
        return cron
    }
}
